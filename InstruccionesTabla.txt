CREATE DATABASE IF NOT EXISTS cars;
USE cars;

CREATE TABLE IF NOT EXISTS vehicles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    placacar VARCHAR(10) NOT NULL,
    modelo VARCHAR(50) NOT NULL,
    anho INT NOT NULL,
    binauto VARCHAR(17)NOT NULL
);
